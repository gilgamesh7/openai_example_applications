# Setup on VDI
- poetry init
- poetry config virtualenvs.in-project true
- poetry install
    - poetry add
    - poetry remove
- on python shell
    - MacOS : export OPENAI_API_KEY=
    - Powershell : $env:OPENAI_API_KEY=
